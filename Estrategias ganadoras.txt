        self._mr_exit = int(self.get_parameter("mr_exit", 50))
        self._mr_alloc = float(self.get_parameter("mr_alloc", 0.25))


        # Scheduled events: stop check -> rebalance
        self.schedule.on(self.date_rules.every_day(), self.time_rules.midnight, self._check_portfolio_stop)
        self.schedule.on(self.date_rules.every_day(), self.time_rules.midnight, self._rebalance)


    def _check_portfolio_stop(self):
        current = self.portfolio.total_portfolio_value
        if current > self._portfolio_peak:
            self._portfolio_peak = current
            self._stop_counter = 0


        if self._in_stop_mode:
            self._cooldown_days += 1
            if self._cooldown_days >= self._cooldown_after_stop:
                self._in_stop_mode = False
                self._cooldown_days = 0
            return

            dd = (self._portfolio_peak - current) / self._portfolio_peak

                self._stop_counter += 1
                if self._stop_counter >= self._max_stop_cycles:
                    self._portfolio_peak = current
                    self._stop_counter = 0
                    self.debug(f"PEAK RESET after {self._max_stop_cycles} stop cycles, new peak=${current:,.0f}")
                else:
                    self._in_stop_mode = True
                    self._cooldown_days = 0
                    self.liquidate()
                    self.debug(f"STOP {self._stop_counter}/{self._max_stop_cycles}: DD={dd:.2%}, Portfolio=${current:,.0f}")

            target_syms = {self._gld: 0.5 * lev, self._shv: 0.5 * lev}

            need_trade = True

            self.set_holdings(targets, liquidate_existing_holdings=True)
        else:
            self.debug(f"Rebalance: TQQQ={tqqq_ok} ETH={eth_ok} GLD={gld_ok} -> skip")

        self._tqqq = self.add_equity('TQQQ', Resolution.DAILY).symbol
        self._eth = self.add_crypto('ETHUSD', Resolution.DAILY).symbol
        self._btc = self.add_crypto('BTCUSD', Resolution.DAILY).symbol

        self._gld = self.add_equity('GLD', Resolution.DAILY).symbol

        self._max_lev = float(self.get_parameter("max_lev", 2.0))

        self.schedule.on(self.date_rules.every_day(), self.time_rules.midnight, self._check_portfolio_stop)

    def _track_return(self):
        current = self.portfolio.total_portfolio_value
        if self._prev_equity > 0:
            ret = (current - self._prev_equity) / self._prev_equity
            self._daily_returns.append(ret)
            if len(self._daily_returns) > self._vol_lookback:
                self._daily_returns.pop(0)
        self._prev_equity = current


    def _compute_leverage(self):
        if len(self._daily_returns) < self._vol_lookback:
            return 1.0
        recent_vol = np.std(self._daily_returns)
        if recent_vol <= 0:
            return 1.0
        ann_vol = recent_vol * np.sqrt(252)
        lev = self._target_vol / ann_vol
        lev = max(0.5, min(lev, self._max_lev))
        # Reduce leverage after recent stop to prevent death spiral
        if self._days_since_stop < 10:
            lev = min(lev, 1.0)
        return lev

                else:

                    self._cooldown_days = 0
                    self._days_since_stop = 0
